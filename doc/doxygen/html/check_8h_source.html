1436  const char* _ck_y_s; \

1438  const char* _ck_y_q; \

1440  _ck_x_q = "\""; \
1441  _ck_x_s = _ck_x; \
1442  } else { \
1443  _ck_x_q = ""; \
1444  _ck_x_s = "(null)"; \
1445  } \
1446  if (_ck_y != NULL) { \

1451  _ck_y_s = "(null)"; \
